## Step 2: Container Styling Add Karna

**Soch:** "Form ko ek styled container mein wrap karna hai — border, padding, rounded corners ke saath."

**Code:**
```tsx
<div className="border-x-4 border-gray-600 p-10 w-170 mx-auto mt-10 rounded-2xl">
  <h1 className="text-2xl font-bold mb-4 text-center text-cyan-600">
    Login Form
  </h1>
  <form onSubmit={handleSubmit(onSubmit)}>
    {/* Form fields */}
  </form>

```

**Explanation:**
- `border-x-4 border-gray-600` — Left/right borders (gray color)
- `p-10` — Padding (40px)
- `w-170` — Width (680px)
- `mx-auto` — Center horizontally
- `mt-10` — Margin top (40px)
- `rounded-2xl` — Rounded corners

## Step 3: Input Fields Styling

**Soch:** "Input fields ko style karna hai — borders, focus states, colors."

**Code:**
```tsx
<div className="mb-4">
  <label htmlFor="email" className="block text-sm font-medium text-gray-400">
    Email:
  </label>
  <input
    className="border-y border-gray-300 rounded-md shadow-sm focus:outline-none focus:ring-cyan-500 focus:border-cyan-500 w-full px-3 py-1"

    id="email"
    placeholder="Enter your email"

  {CSerrors.email && (
    <span className="text-red-500">{CSerrors.email.message}</span>
  )}
</div>
```

**Explanation:**
- `border-y border-gray-300` — Top/bottom borders (gray)
- `rounded-md` — Slightly rounded corners
- `shadow-sm` — Small shadow
- `focus:outline-none focus:ring-cyan-500 focus:border-cyan-500` — Focus state (cyan ring)
- `w-full` — Full width
- `px-3 py-1` — Padding (horizontal 12px, vertical 4px)
- `text-gray-400` — Label text color (light gray)
- `text-red-500` — Error text color (red)

## Step 4: Submit Button Styling

**Soch:** "Submit button ko style karna hai — hover effects, disabled state."

**Code:**
```tsx
<button
  type="submit"
  disabled={isPending}
  className={`w-full py-2 px-4 rounded-md text-white cursor-pointer ${
    isPending
      ? "bg-gray-400 cursor-not-allowed"
      : "bg-cyan-600 hover:bg-cyan-700"
  }`}
>
  {isPending ? "Logging in..." : "Login"}
</button>
```

**Explanation:**
- `w-full` — Full width
- `py-2 px-4` — Padding (vertical 8px, horizontal 16px)
- `rounded-md` — Rounded corners
- `text-white` — White text
- `bg-cyan-600` — Cyan background (normal state)
- `hover:bg-cyan-700` — Darker cyan on hover
- `bg-gray-400 cursor-not-allowed` — Gray background when disabled
- `{isPending ? "Logging in..." : "Login"}` — Dynamic text

---

## Step 5: `isPending` State — Loading Indicator

**Soch:** "Form submit hone par button disable karna hai aur 'Logging in...' dikhana hai."

**`isPending` kahan se aata hai:**
- `useActionState` se aata hai (register page mein hai)
- Form submit hone par `true` ho jata hai
- Server action complete hone par `false` ho jata hai

**Note:** Abhi humne `useActionState` use nahi kiya (baad mein aayega). Filhal `isPending` ko `false` set kar sakte hain ya `useActionState` add kar sakte hain.

## Step 6: Complete Code

```tsx
// app/(auth)/login/page.tsx

"use client";

import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { loginSchema, LoginInput } from "@/lib/validations/auth";

export default function LoginPage() {
  const {
    register,
    handleSubmit,
    formState: { errors: CSerrors },
  } = useForm<LoginInput>({
    resolver: zodResolver(loginSchema),
  });

  const onSubmit = (data: LoginInput) => {
    console.log("Form submitted:", data);
    // Baad mein yahan server action call karenge
  };

  return (
    <div className="border-x-4 border-gray-600 p-10 w-170 mx-auto mt-10 rounded-2xl">
      <h1 className="text-2xl font-bold mb-4 text-center text-cyan-600">
        Login Form
      </h1>

      <form onSubmit={handleSubmit(onSubmit)}>
        <div className="mb-4">
          <label
            htmlFor="email"
            className="block text-sm font-medium text-gray-400"
          >
            Email:
          </label>
          <input
            className="border-y border-gray-300 rounded-md shadow-sm focus:outline-none focus:ring-cyan-500 focus:border-cyan-500 w-full px-3 py-1"
            type="email"
            id="email"
            placeholder="Enter your email"
            {...register("email")}
          />
          {CSerrors.email && (
            <span className="text-red-500">{CSerrors.email.message}</span>
          )}
        </div>

        <div className="mb-4">
          <label
            htmlFor="password"
            className="block text-sm font-medium text-gray-400"
          >
            Password:
          </label>
          <input
            className="border-y border-gray-300 rounded-md shadow-sm focus:outline-none focus:ring-cyan-500 focus:border-cyan-500 w-full px-3 py-1"
            type="password"
            id="password"
            placeholder="Enter your password"
            {...register("password")}
          />
          {CSerrors.password && (
            <span className="text-red-500">{CSerrors.password.message}</span>
          )}
        </div>

        <button
          type="submit"
          className="w-full py-2 px-4 rounded-md text-white cursor-pointer bg-cyan-600 hover:bg-cyan-700"
        >
          Login
        </button>
      </form>
    </div>
  );
}
```
