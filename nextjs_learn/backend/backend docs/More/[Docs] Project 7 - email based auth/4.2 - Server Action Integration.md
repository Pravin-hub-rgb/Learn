## Step 4: Problem — Error Messages Kaise Dikhayein?

**Soch:** "Direct `loginUser(formData)` call karne se response capture nahi ho raha. Mujhe server action ka response (`{ error: "...", success: true }`) capture karke UI par dikhana hai."

**Solution:** `useActionState` hook use karna — yeh server action ka response capture karta hai.

## Step 5: `useActionState` Hook Introduce Karna

**Soch:** "`useActionState` hook use karunga jo server action ka response capture karega aur state manage karega."

**`useActionState` kya hai:**
- React ka hook — server actions ke saath use karne ke liye
- Server action ka response capture karta hai
- `isPending` state deta hai (loading indicator ke liye)

**Syntax:**
```tsx
const [state, formAction, isPending] = useActionState(serverAction, initialState);
```

**Return values:**
- `state` — Server action ka response (jo return hota hai)
- `formAction` — Form ko isse connect karte hain
- `isPending` — Boolean — `true` jab action chal raha ho

## Step 12: Complete Code — Login Page

```tsx
// app/(auth)/login/page.tsx

"use client";

import { startTransition, useActionState } from "react";
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { loginSchema, LoginInput } from "@/lib/validations/auth";
import { loginUser } from "@/app/actions/auth";

export default function LoginPage() {
  const {
    register,
    handleSubmit,
    formState: { errors: CSerrors },
  } = useForm<LoginInput>({
    resolver: zodResolver(loginSchema),
  });

  const [state, formAction, isPending] = useActionState(loginUser, {
    error: "",
    success: false,
  });

  const onSubmit = (data: LoginInput) => {
    const formData = new FormData();
    formData.append("email", data.email);
    formData.append("password", data.password);
    startTransition(() => {
      formAction(formData);
    });
  };

  return (
    <div className="border-x-4 border-gray-600 p-10 w-170 mx-auto mt-10 rounded-2xl">
      <h1 className="text-2xl font-bold mb-4 text-center text-cyan-600">
        Login Form
      </h1>

      <form onSubmit={handleSubmit(onSubmit)}>
        <div className="mb-4">
          <label
            htmlFor="email"
            className="block text-sm font-medium text-gray-400"
          >
            Email:
          </label>
          <input
            className="border-y border-gray-300 rounded-md shadow-sm focus:outline-none focus:ring-cyan-500 focus:border-cyan-500 w-full px-3 py-1"
            type="email"
            id="email"
            placeholder="Enter your email"
            {...register("email")}
          />
          {CSerrors.email && (
            <span className="text-red-500">{CSerrors.email.message}</span>
          )}
        </div>

        <div className="mb-4">
          <label
            htmlFor="password"
            className="block text-sm font-medium text-gray-400"
          >
            Password:
          </label>
          <input
            className="border-y border-gray-300 rounded-md shadow-sm focus:outline-none focus:ring-cyan-500 focus:border-cyan-500 w-full px-3 py-1"
            type="password"
            id="password"
            placeholder="Enter your password"
            {...register("password")}
          />
          {CSerrors.password && (
            <span className="text-red-500">{CSerrors.password.message}</span>
          )}
        </div>

        <button
          type="submit"
          disabled={isPending}
          className={`w-full py-2 px-4 rounded-md text-white cursor-pointer ${
            isPending
              ? "bg-gray-400 cursor-not-allowed"
              : "bg-cyan-600 hover:bg-cyan-700"
          }`}
        >
          {isPending ? "Logging in..." : "Login"}
        </button>

        {state?.error && (
          <div className="text-red-500 mt-4">{state.error}</div>
        )}
        {state?.success && (
          <div className="text-green-500 mt-4">Login successful!</div>
        )}
      </form>
    </div>
  );
}
```
